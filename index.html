relational and logical operators:

relational operators

num1 <- 100
num2 <- 200
num3 <- 100

num1 >  num2         # is NUM1 greater than NUM2?
num1 <  num2         # is NUM1 less than NUM2?
num1 >= num3         # is NUM1 greater than or equal to NUM3?
num3 <= num2         # is NUM3 less than or equal to NUM2?
num1 == num3         # is NUM1 equal to NUM3?
num1 != num2         # is NUM1 not equal to NUM2? 
!(num1 == num3)      # is NUM1 not equal to NUM2?
num1 + num3 == num2  # equal to

logical operators

num1 <- 100
num2 <- 200
num3 <- 100

num1 <  num2 & num1 <= num3   # AND operator
num1 == num2 | num1 == num3   # OR operator

e.g., compute Future Value

Future Value formula

int <- 0.08 # 8% annual rate
nterms <- 7 # number of years
amount <- 1000 # present value

# compute future value
fv <- amount * ((1 + int)^nterms); fv


[1] 1713.824

e.g., compute Future Value

Future Value formula

# modular calculations 

fv1 <- (1 + int)^nterms; # step 1

fv1 <- fv1 * amount; fv1 # step 2


[1] 1713.824

e.g., compute Future Value for a vector of interest rates

Future Value formula

int1 <- seq(0.02, 0.14, by = 0.02); int1
